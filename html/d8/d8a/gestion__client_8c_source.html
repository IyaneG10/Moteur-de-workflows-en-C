306 char * getAdresseClient(long socket)
307 {
308  char * nomMachine = malloc(SIZE_NOM_MACHINE);
309  struct sockaddr_storage adresse;
310  struct sockaddr *padresse=(struct sockaddr *)&adresse;
311  socklen_t taille=sizeof adresse;
312  int statut;
313  struct hostent *machine;
314 
315  /* Recupere l'adresse de la socket distante */
316  statut=getpeername(socket,padresse,&taille);
317  if(statut<0){
318  perror("getAdresseClient.getpeername");
319  exit(-1);
320  }
321 
322  /* Recupere le nom de la machine */
323  void *ip;
324  int taille_ip;
325  int taille_nom;
326  if(padresse->sa_family==AF_UNSPEC){
327  struct sockaddr_in *ipv4=(struct sockaddr_in *)padresse;
328  ip=(void *)&ipv4->sin_addr;
329  taille_ip=sizeof(ipv4->sin_addr);
330  taille_nom=INET_ADDRSTRLEN;
331  }
332 
333  if(padresse->sa_family==AF_INET6){
334  struct sockaddr_in6 *ipv6=(struct sockaddr_in6 *)padresse;
335  ip=(void *)&ipv6->sin6_addr;
336  taille_ip=sizeof(ipv6->sin6_addr);
337  taille_nom=INET6_ADDRSTRLEN;
338  }
339 
340  machine=gethostbyaddr(ip,taille_ip,padresse->sa_family);
341 
342  if(machine==NULL){
343  inet_ntop(padresse->sa_family,ip,nomMachine,taille_nom);
344  exit(-1);
345  }
346  else{
347  strncpy(nomMachine,machine->h_name,LONG_ID);
348  }
349 
350  return nomMachine;
351 }
